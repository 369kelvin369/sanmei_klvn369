// 用語ツールチップ
// ============================================================
function Term({ word, children }) {
  const [show, setShow] = useState(false);
  const [pos, setPos] = useState(null); // {left, top, arrowLeft, placement}
  const anchorRef = React.useRef(null);
  const def = GLOSSARY[word];

  // 表示位置を計算（画面内に必ず収まるようクランプ）
  const computePos = React.useCallback(() => {
    const el = anchorRef.current;
    if (!el) return;
    const rect = el.getBoundingClientRect();
    const vw = window.innerWidth;
    const margin = 12; // 画面端からの最低マージン
    const tipW = Math.min(280, vw - margin * 2); // 吹き出し幅（画面が狭ければ縮める）
    const anchorCenter = rect.left + rect.width / 2;

    // 中央寄せを基本に、左右が画面からはみ出さないようにクランプ
    let left = anchorCenter - tipW / 2;
    if (left < margin) left = margin;
    if (left + tipW > vw - margin) left = vw - margin - tipW;

    // 矢印は実際のアンカー中心を指すように（吹き出し内の相対位置）
    let arrowLeft = anchorCenter - left;
    arrowLeft = Math.max(16, Math.min(tipW - 16, arrowLeft));

    // 基本は上に出すが、上に余白がなければ下に出す
    const placement = rect.top > 160 ? "top" : "bottom";
    const top = placement === "top" ? rect.top - 8 : rect.bottom + 8;

    setPos({ left, top, width: tipW, arrowLeft, placement });
  }, []);

  const handleOpen = () => {
    computePos();
    setShow(true);
  };
  const handleClose = () => setShow(false);
  const handleToggle = () => {
    if (show) {
      setShow(false);
    } else {
      computePos();
      setShow(true);
    }
  };

  // 開いている間にスクロール／リサイズされたら位置を追従、画面外タップで閉じる
  React.useEffect(() => {
    if (!show) return;
    const reposition = () => computePos();
    window.addEventListener("scroll", reposition, true);
    window.addEventListener("resize", reposition);
    return () => {
      window.removeEventListener("scroll", reposition, true);
      window.removeEventListener("resize", reposition);
    };
  }, [show, computePos]);

  return (
    <span
      ref={anchorRef}
      style={{ position: "relative", display: "inline" }}
      onMouseEnter={handleOpen}
      onMouseLeave={handleClose}
      onClick={handleToggle}
    >
      <span
        style={{
          borderBottom: "1.5px dotted #b03a2e",
          cursor: "help",
          color: "#8b2e22",
        }}
      >
        {children || word}
      </span>
      {show && def && pos && (
        <span
          style={{
            position: "fixed",
            left: pos.left + "px",
            top: pos.top + "px",
            transform: pos.placement === "top" ? "translateY(-100%)" : "none",
            width: pos.width + "px",
            background: "#2b2420",
            color: "#f4ecd8",
            padding: "10px 14px",
            borderRadius: "8px",
            fontSize: "12.5px",
            lineHeight: "1.7",
            zIndex: 9999,
            boxShadow: "0 8px 24px rgba(0,0,0,0.35)",
            fontWeight: "normal",
            textAlign: "left",
            letterSpacing: "0.02em",
            pointerEvents: "none",
          }}
        >
          {def}
          {/* 矢印（アンカーを指す三角） */}
          <span
            style={{
              position: "absolute",
              left: pos.arrowLeft + "px",
              transform: "translateX(-50%)",
              [pos.placement === "top" ? "top" : "bottom"]: "100%",
              width: 0,
              height: 0,
              borderLeft: "6px solid transparent",
              borderRight: "6px solid transparent",
              [pos.placement === "top" ? "borderTop" : "borderBottom"]:
                "7px solid #2b2420",
            }}
          />
        </span>
      )}
    </span>
  );
}

  const [loaded, setLoaded] = useState(false);

  // 入力state
  const [name, setName] = useState("");
  const [birth, setBirth] = useState("");
  const [tenchusatsu, setTenchusatsu] = useState("");
  // 人体図の5星
  const [body, setBody] = useState({
    head: "禄存星",
    rightHand: "鳳閣星",
    chest: "玉堂星",
    leftHand: "天恍星",
    belly: "貫索星",
  });
  // 大運（最大8行）
  const [daiun, setDaiun] = useState([
    { age: "2", kanshi: "庚辰", main: "石門星", follow: "天庫星" },
    { age: "12", kanshi: "己卯", main: "龍高星", follow: "天馳星" },
    { age: "22", kanshi: "戊寅", main: "玉堂星", follow: "天報星" },
    { age: "32", kanshi: "丁丑", main: "車騎星", follow: "天印星" },
    { age: "42", kanshi: "丙子", main: "牽牛星", follow: "天貴星" },
  ]);
  const [result, setResult] = useState(null);

  // --- 保存データの読み込み ---
  useEffect(() => {
    (async () => {
      try {
        const r = await window.storage.get("sanmei:profile");
        if (r && r.value) {
          const d = JSON.parse(r.value);
          setSaved(d);
          if (d.name) setName(d.name);
          if (d.birth) setBirth(d.birth);
          if (d.tenchusatsu) setTenchusatsu(d.tenchusatsu);
          if (d.body) setBody(d.body);
          if (d.daiun) setDaiun(d.daiun);
          if (d.result) setResult(d.result);
        }
      } catch (e) {
        // 未保存
      }
      setLoaded(true);
    })();
  }, []);

  const handleSave = async () => {
    const data = { name, birth, tenchusatsu, body, daiun, result };
    try {
      await window.storage.set("sanmei:profile", JSON.stringify(data));
      setSaved(data);
      alert("鑑定結果を保存しました。次回開いたときに自動で読み込まれます。");
    } catch (e) {
      alert("保存に失敗しました。");
    }
  };

  const handleReset = async () => {
    if (!confirm("保存データを消去しますか？")) return;
    try {
      await window.storage.delete("sanmei:profile");
    } catch (e) {}
    setSaved(null);
  };

  const runKantei = () => {
    setResult({ name, birth, tenchusatsu, body: { ...body }, daiun: [...daiun] });
    window.scrollTo({ top: 9999, behavior: "smooth" });
  };

  const updateDaiun = (i, field, val) => {
    setDaiun((d) => d.map((row, idx) => (idx === i ? { ...row, [field]: val } : row)));
  };
  const addDaiun = () => setDaiun((d) => [...d, { age: "", kanshi: "", main: "貫索星", follow: "天禄星" }]);
  const removeDaiun = (i) => setDaiun((d) => d.filter((_, idx) => idx !== i));

  return (
    <div style={S.page}>
      <style>{globalCSS}</style>

      {/* ヘッダー */}
      <header style={S.header}>
        <div style={S.headerInner}>
          <div style={S.seal}>算<br/>命</div>
          <div>
            <h1 style={S.title}>算命学 鑑定 ＆ 学習ノート</h1>
            <p style={S.subtitle}>計算済みの結果を入力して、宿命をひもとく</p>
          </div>
        </div>
      </header>

      {/* タブ */}
      <nav style={S.tabs}>
        {[
          ["kantei", "鑑定"],
          ["gakushu", "星を学ぶ"],
          ["aisho", "相性"],
          ["nenun", "時期をみる"],
        ].map(([k, label]) => (
          <button
            key={k}
            onClick={() => setTab(k)}
            style={{ ...S.tab, ...(tab === k ? S.tabActive : {}) }}
          >
            {label}
          </button>
        ))}
      </nav>

      <main style={S.main}>
        {tab === "kantei" && (
          <KanteiTab
            {...{ name, setName, birth, setBirth, tenchusatsu, setTenchusatsu,
              body, setBody, daiun, updateDaiun, addDaiun, removeDaiun,
              runKantei, result, handleSave, handleReset, saved }}
          />
        )}
        {tab === "gakushu" && <GakushuTab />}
        {tab === "aisho" && <AishoTab />}
        {tab === "nenun" && <NenunTab birth={birth} setBirth={setBirth} tenchusatsu={tenchusatsu} setTenchusatsu={setTenchusatsu} daiun={daiun} />}
      </main>

      <footer style={S.footer}>
        <p style={{ margin: 0 }}>
          ※ このツールは計算機能を持ちません。
          <a href="https://www.shugakuin.co.jp/fate_calculation" target="_blank" rel="noreferrer" style={S.link}>朱学院の無料算出ページ</a>
          などで算出した結果を入力してお使いください。
        </p>
        <p style={{ margin: "6px 0 0", opacity: 0.7 }}>占いは人生を楽しむためのもの。最後はご自身の意志で。</p>
      </footer>
    </div>
  );
}

function ResultView({ result }) {
  const { name, birth, tenchusatsu, body, daiun } = result;
  const centerStar = body.chest;

  // 本能バランス集計
  const honnoCount = {};
  Object.values(body).forEach((star) => {
    for (const [g, info] of Object.entries(HONNO_GROUPS)) {
      if (info.stars.includes(star)) honnoCount[g] = (honnoCount[g] || 0) + 1;
    }
  });

  return (
    <div style={S.resultWrap} id="result">
      <div style={S.resultHeader}>
        <span style={S.resultBadge}>鑑定結果</span>
        <h2 style={S.resultName}>{name || "あなた"}さんの宿命</h2>
        {birth && <p style={S.resultBirth}>{birth}</p>}
      </div>

      {/* 人体図 */}
      <section>
        <h3 style={S.sectionTitle}>◆ 人体図</h3>
        <p style={S.help}>体の各部位に星を配置した図です。中央の「胸」があなたの中心となる本質です。</p>
        <BodyChart body={body} />
      </section>

      {/* 中心星の解説 */}
      <section>
        <h3 style={S.sectionTitle}>◆ あなたの中心の星：{centerStar}</h3>
        <StarDetail name={centerStar} highlight />
      </section>

      {/* 各部位の星 */}
      <section>
        <h3 style={S.sectionTitle}>◆ それぞれの場所の星</h3>
        <div style={S.starList}>
          {BODY_POSITIONS.filter((p) => !p.center).map((pos) => (
            <div key={pos.key} style={S.starListItem}>
              <div style={S.starListLabel}>{pos.label}</div>
              <StarDetail name={body[pos.key]} compact />
            </div>
          ))}
        </div>
      </section>

      {/* 本能バランス */}
      <section>
        <h3 style={S.sectionTitle}>◆ あなたの本能バランス</h3>
        <p style={S.help}>
          算命学では心の働きを5つの<Term word="本能">本能</Term>に分けます。
          あなたの星がどの本能に偏っているかで、性質の傾向がわかります。
        </p>
        <div style={S.honnoBars}>
          {Object.entries(HONNO_GROUPS).map(([g, info]) => {
            const c = honnoCount[g] || 0;
            return (
              <div key={g} style={S.honnoBarRow}>
                <span style={S.honnoBarLabel}>{g}</span>
                <div style={S.honnoBarTrack}>
                  <div style={{ ...S.honnoBarFill, width: `${(c / 5) * 100}%`, opacity: c ? 1 : 0.15 }} />
                </div>
                <span style={S.honnoBarCount}>{c}</span>
              </div>
            );
          })}
        </div>
        <p style={S.honnoNote}>
          {(() => {
            const top = Object.entries(honnoCount).sort((a, b) => b[1] - a[1])[0];
            if (!top) return "";
            return `あなたは「${top[0]}」の星が多めです。${HONNO_GROUPS[top[0]].desc}この力を活かす生き方が、あなたらしさにつながります。`;
          })()}
        </p>
      </section>

      {/* 天中殺 */}
      {tenchusatsu && (
        <section>
          <h3 style={S.sectionTitle}>◆ 天中殺：{tenchusatsu}天中殺</h3>
          <TenchusatsuDetail type={tenchusatsu} />

      {/* 大運 */}

        </p>
        <DaiunFlow daiun={daiun} />
      </section>
    </div>
  );

  // 五行の相生・相剋による簡易相性判定

  const ia = ORDER.indexOf(ga), ib = ORDER.indexOf(gb);
  let relation, relColor, relDesc;

  } else if ((ib + 1) % 5 === ia) {

    relDesc = `「${gb}」は「${ga}」を生み育てる関係（相生）。相手があなたを支え、伸ばしてくれます。甘えすぎず感謝を忘れずに。`;

    relDesc = `お互いを律し、緊張感を生む関係（相剋）。ぶつかりやすいですが、その刺激が成長を促すことも。違いを尊重できれば、最高の好敵手・パートナーになります。`;

  return (

        </p>
        <div style={S.aishoSelectRow}>
          <select style={S.aishoSelect} value={a} onChange={(e) => setA(e.target.value)}>
            {stars.map((s) => <option key={s}>{s}</option>)}
          </select>
          <span style={S.aishoVs}>×</span>
          <select style={S.aishoSelect} value={b} onChange={(e) => setB(e.target.value)}>
            {stars.map((s) => <option key={s}>{s}</option>)}
          </select>
        </div>

        <div style={{ ...S.aishoResult, borderColor: relColor }}>
          <div style={{ ...S.aishoRelation, color: relColor }}>{relation}</div>
          <p style={S.aishoDesc}>{relDesc}</p>
        </div>

        <div style={S.aishoStars}>
          <StarDetail name={a} compact />
          <StarDetail name={b} compact />
        </div>
      </Card>

      <Card title="五行の相性ってなに？">
        <p style={S.proseP}>
          算命学のベースには<b>五行（ごぎょう）</b>という考え方があります。
          世界は「木・火・土・金・水」の5つのエネルギーでできているという東洋哲学です。
          これが5つの<Term word="本能">本能</Term>に対応しています。
        </p>
        <p style={S.proseP}>

          <b>・相生（そうじょう）</b>＝生み育てる、助け合う良い関係（例：木は燃えて火を生む）<br/>
          <b>・相剋（そうこく）</b>＝抑え合う、刺激し合う関係（例：水は火を消す）

// 十干・十二支（暦で固定。計算がズレない部分）
const STEMS = ["甲", "乙", "丙", "丁", "戊", "己", "庚", "辛", "壬", "癸"];
const STEM_YOMI = { 甲: "きのえ", 乙: "きのと", 丙: "ひのえ", 丁: "ひのと", 戊: "つちのえ", 己: "つちのと", 庚: "かのえ", 辛: "かのと", 壬: "みずのえ", 癸: "みずのと" };
const BRANCHES = ["子", "丑", "寅", "卯", "辰", "巳", "午", "未", "申", "酉", "戌", "亥"];
const BRANCH_YOMI = { 子: "ね", 丑: "うし", 寅: "とら", 卯: "う", 辰: "たつ", 巳: "み", 午: "うま", 未: "ひつじ", 申: "さる", 酉: "とり", 戌: "いぬ", 亥: "い" };

// 天中殺ペア（その2支がその人の充電期間）
const TENCHU_PAIRS = {
  子丑: ["子", "丑"], 寅卯: ["寅", "卯"], 辰巳: ["辰", "巳"],
  午未: ["午", "未"], 申酉: ["申", "酉"], 戌亥: ["戌", "亥"],
};

// 年の干支を計算（西暦→干支）。立春前は注意（後述の注記で補足）
function yearGanshi(year) {
  const sIdx = ((year - 4) % 10 + 10) % 10;
  const bIdx = ((year - 4) % 12 + 12) % 12;
  const stem = STEMS[sIdx];
  const branch = BRANCHES[bIdx];
  return { stem, branch, text: stem + branch };
}

// 十二支ごとの一般的な「その年の気質」（干支暦のイメージ。個人の宿命とは別の全体運）
const BRANCH_THEME = {
  子: { season: "冬の始まり", word: "種まき・スタート", desc: "新しいサイクルの始まり。表には見えないところで物事が動き出す、仕込み・準備の年。" },
  丑: { season: "晩冬", word: "耐える・育てる", desc: "じっと耐えて力を蓄える年。地道な努力が後で芽吹く、我慢と土台づくりの時期。" },
  寅: { season: "早春", word: "動き出す・発芽", desc: "蓄えた力が動き出す年。新しいことを始める・行動に移すのに勢いがつく時期。" },
  卯: { season: "春", word: "成長・伸びる", desc: "芽が伸びるように物事が育つ年。広がり・成長・人間関係の発展が期待できる時期。" },
  辰: { season: "晩春", word: "変化・うねり", desc: "大きく動き変化する年。波が立ちやすく、飛躍もあればブレもある、ダイナミックな時期。" },
  巳: { season: "初夏", word: "充実・知恵", desc: "内面が充実し知恵が冴える年。じっくり考え、内なる力を高めるのに向く時期。" },
  午: { season: "夏（盛り）", word: "最高潮・発展", desc: "エネルギーが最高潮に達する年。活発に表に出て、勝負・発展に向く明るい時期。" },
  未: { season: "晩夏", word: "成熟・調整", desc: "盛りを過ぎ成熟へ向かう年。まとめ・調整・人との和を大切にすると良い時期。" },
  申: { season: "初秋", word: "展開・実りへ", desc: "実りへ向けて物事が展開する年。器用に立ち回り、成果につなげていく時期。" },
  酉: { season: "秋", word: "収穫・完成", desc: "努力が形になる収穫の年。結果が出る・仕上げる・けじめをつけるのに向く時期。" },
  戌: { season: "晩秋", word: "締めくくり・整理", desc: "一区切りつける年。守りを固め、整理し、次に備えて身辺を整える時期。" },
  亥: { season: "冬", word: "蓄える・休息", desc: "サイクルの終わりに力を蓄える年。休息・充電し、次の始まりに備える時期。" },
};

// 天中殺の年の読み（その人にとって最も確実に読める部分）
function nenunReading(year, tenchusatsu) {
  const { branch } = yearGanshi(year);
  const pair = TENCHU_PAIRS[tenchusatsu];
  const theme = BRANCH_THEME[branch];
  if (pair) {
    if (branch === pair[0]) {
      return {
        type: "tenchu1",
        label: "天中殺 1年目",
        color: "#b03a2e",
        text: "あなたの天中殺の1年目。運気の充電期間に入ります。新しいことを始めたり、大きな決断をするより、これまでの整理・学び直し・人とのご縁を育てることに使うと吉。焦らず、流れに逆らわないのがコツです。",
      };
    }
    if (branch === pair[1]) {
      return {
        type: "tenchu2",
        label: "天中殺 2年目",
        color: "#b03a2e",
        text: "天中殺の2年目（最終年）。引き続き種まき・準備の時期です。結婚・転職・独立・引っ越し・大きな契約などの『人生の方向を変える決断』は、できればこの時期を避けるか、慎重に。ここで耐えたぶん、明けてから動きやすくなります。",
      };
    }
    // 天中殺明け（2年目の次の支）
    const after = BRANCHES[(BRANCHES.indexOf(pair[1]) + 1) % 12];
    if (branch === after) {
      return {
        type: "ake",
        label: "天中殺 明け 🌅",
        color: "#6b7a35",
        text: "天中殺が明けた最初の年。長い充電期間を抜け、新しいサイクルがスタートします。仕切り直し・再スタート・新しい挑戦に向く、追い風の年。天中殺中に温めてきたことを、ここから形にしていきましょう。",
      };
    }
  }
  return {
    type: "normal",
    label: theme.word,
    color: "#a8862c",
    text: theme.desc,
  };
}

function NenunTab({ birth, setBirth, tenchusatsu, setTenchusatsu, daiun }) {
  const thisYear = new Date().getFullYear();
  // 生年を birth 文字列から推定（なければ空）
  const guessedYear = (() => {
    const m = (birth || "").match(/(\d{4})/);
    return m ? parseInt(m[1], 10) : "";
  })();
  const [birthYear, setBirthYear] = useState(guessedYear || "");
  const [selYear, setSelYear] = useState(thisYear);

  const hasTenchu = !!TENCHU_PAIRS[tenchusatsu];
  const age = birthYear ? thisYear - parseInt(birthYear, 10) : null;

  // 選択年の鑑定
  const gs = yearGanshi(selYear);
  const reading = nenunReading(selYear, tenchusatsu);
  const theme = BRANCH_THEME[gs.branch];

  // タイムライン（前年〜10年先）
  const years = [];
  for (let y = thisYear - 1; y <= thisYear + 10; y++) years.push(y);

  // 現在の大運（年齢から）
  const currentDaiun = (() => {
    if (age == null || !daiun || !daiun.length) return null;
    let cur = null;
    for (const row of daiun) {
      const a = parseInt(row.age, 10);
      if (!isNaN(a) && age >= a) cur = row;
    }
    return cur;
  })();

  return (
    <div>
      <Card title="時期をみる ― 年運と天中殺" step>
        <p style={S.help}>
          <Term word="大運">大運</Term>が10年の「季節」なら、年運はその年その年の「天気」。
          ここでは<b>暦で確実に決まる「年の干支」と「天中殺の年」</b>を自動で判定します。
        </p>

        <div style={S.nenunInputRow}>
          <div style={{ flex: 1 }}>
            <label style={S.label}>生まれ年（西暦）</label>
            <input
              style={S.input}
              value={birthYear}
              onChange={(e) => setBirthYear(e.target.value.replace(/[^0-9]/g, ""))}
              placeholder="例：1995"
              inputMode="numeric"
            />
          </div>
          <div style={{ flex: 1 }}>
            <label style={S.label}><Term word="天中殺">天中殺</Term></label>
            <select style={S.input} value={tenchusatsu} onChange={(e) => setTenchusatsu(e.target.value)}>
              <option value="">未選択</option>
              {Object.keys(TENCHU_PAIRS).map((t) => (
                <option key={t} value={t}>{t}天中殺</option>
              ))}
            </select>
          </div>
        </div>

        {!hasTenchu && (
          <p style={S.nenunWarn}>
            ※ 天中殺を選ぶと、その年があなたの「充電期間」かどうかも判定できます。
            （鑑定タブで入力した天中殺はここにも反映されます）
          </p>
        )}

        {/* 立春の注意 */}
        <div style={S.rissyunNote}>
          <b>📌 ひとつだけ注意</b><br />
          算命学の「1年」は1月1日ではなく<b>立春（2月4日ごろ）から</b>始まります。
          そのため<b>1月〜2月初旬</b>については、運気の上では前年として扱う場合があります。
          その時期の判定はこのズレにご注意ください。
        </div>
      </Card>

      {/* タイムライン */}
      <Card title="これからの流れ（年表）">
        <p style={S.help}>
          年をタップすると下に詳しい解説が出ます。
          <span style={{ color: vermillion, fontWeight: 700 }}>赤い年</span>があなたの天中殺、
          <span style={{ color: moss, fontWeight: 700 }}>緑の年</span>が天中殺明けです。
        </p>
        <div style={S.timeline}>
          {years.map((y) => {
            const r = nenunReading(y, tenchusatsu);
            const g = yearGanshi(y);
            const isSel = y === selYear;
            const isNow = y === thisYear;
            let bg = "#fff", bd = `${gold}44`, col = ink;
            if (r.type === "tenchu1" || r.type === "tenchu2") { bg = `${vermillion}10`; bd = vermillion; col = vermillion; }
            else if (r.type === "ake") { bg = `${moss}12`; bd = moss; col = moss; }
            return (
              <button
                key={y}
                onClick={() => setSelYear(y)}
                style={{
                  ...S.timeChip,
                  background: bg,
                  borderColor: isSel ? ink : bd,
                  borderWidth: isSel ? 2.5 : 1.5,
                  color: col,
                }}
              >
                {isNow && <span style={S.timeNow}>今</span>}
                <span style={S.timeYear}>{y}</span>
                <span style={S.timeGanshi}>{g.text}</span>
                {(r.type === "tenchu1" || r.type === "tenchu2") && <span style={S.timeTag}>天中殺</span>}
                {r.type === "ake" && <span style={{ ...S.timeTag, color: moss }}>明け</span>}
              </button>
            );
          })}
        </div>
      </Card>

      {/* 選択年の詳細 */}
      <Card title={`${selYear}年（${gs.text}）の運気`}>
        <div style={S.nenunDetailHead}>
          <div style={S.nenunGanshiBox}>
            <span style={S.nenunGanshiText}>{gs.text}</span>
            <span style={S.nenunGanshiYomi}>{STEM_YOMI[gs.stem]}・{BRANCH_YOMI[gs.branch]}</span>
          </div>
          <div>
            <span style={{ ...S.nenunBadge, background: reading.color }}>{reading.label}</span>
            {age != null && (
              <span style={S.nenunAge}>この年あなたは {selYear - parseInt(birthYear, 10)} 歳</span>
            )}
          </div>
        </div>

        <p style={S.nenunReadingText}>{reading.text}</p>

        {/* 一般的な年の気質（天中殺の年でも参考として表示） */}
        {theme && reading.type !== "normal" && (
          <p style={S.nenunSubText}>
            <b>※ {selYear}年そのものの全体傾向：</b>「{theme.season}」にあたり、{theme.word}の年。{theme.desc}
          </p>
        )}

        {/* 大運との重ね合わせ */}
        {currentDaiun && selYear === thisYear && (
          <div style={S.nenunDaiun}>
            <b>🗓 今のあなたの大運（10年の季節）</b><br />
            {currentDaiun.age}歳〜：<b>{currentDaiun.main}</b> の季節
            {TEN_STARS[currentDaiun.main] && `（${TEN_STARS[currentDaiun.main].short}）`}
            <br />
            <span style={{ fontSize: 12, opacity: 0.85 }}>
              この「季節」の中で、上の年運という「天気」が移り変わっていくイメージです。
            </span>
          </div>
        )}
      </Card>

      <Card title="もっと正確に知りたいときは">
        <p style={S.proseP}>
          このタブで出している<b>年の干支・天中殺の年</b>は暦で確実に決まるものなので、まずズレません。
        </p>
        <p style={S.proseP}>
          一方で、<b>「その年があなたにとって何の星（主星・従星）の年か」</b>といった細かい年運は、
          流派や<Term word="十二支">節入り</Term>の計算が絡むため、
          <a href="https://www.shugakuin.co.jp/fate_calculation" target="_blank" rel="noreferrer" style={S.link}>朱学院の算出ページ</a>
          など専門の計算結果を使うのが確実です。出てきた星の意味は「星を学ぶ」タブで調べられます。
        </p>
      </Card>
    </div>
  );
}
